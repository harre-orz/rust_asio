pub struct RecvBufferSize(_);

Socket option for the receive buffer size of a socket.

Implements the SOL_SOCKET/SO_RCVBUF socket option.

Examples

Setting the option:

use asio::*;
use asio::ip::*;
use asio::socket_base::RecvBufferSize;

let io = &IoService::new();
let soc = TcpSocket::new(io, Tcp::v4()).unwrap();

soc.set_option(RecvBufferSize::new(8192)).unwrap();Run

Getting the option:

use asio::*;
use asio::ip::*;
use asio::socket_base::RecvBufferSize;

let io = &IoService::new();
let soc = TcpSocket::new(io, Tcp::v4()).unwrap();

let opt: RecvBufferSize = soc.get_option().unwrap();
let size: usize = opt.get();Run
